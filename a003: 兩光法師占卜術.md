# a003: 兩光法師占卜術

[a003: 兩光法師占卜術](https://zerojudge.tw/ShowProblem?problemid=a003)


~~~cpp
#include<iostream>
using namespace std;

int main()
{
    int m, d, s;
    cin >> m >> d;
    s = (m * 2 + d) % 3;
    if(s == 0)
        cout << "普通";
    else if(s == 1)
        cout << "吉";
    else
        cout << "大吉";
    return 0;
}
