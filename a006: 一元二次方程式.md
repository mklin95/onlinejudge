# a006: 一元二次方程式

[a006: 一元二次方程式](https://zerojudge.tw/ShowProblem?problemid=a006)

~~~cpp
#include<iostream>
#include<math.h>
using namespace std;

int main()
{
    float input;
    int a, b, c, x1, x2;
    cin >> a >> b >> c;
    x1 = (-b + sqrt(b*b - 4*a*c))/(2*a);
    x2 = (-b - sqrt(b*b - 4*a*c))/(2*a);
    if(x1 != x2)
        cout << "Two different roots x1=" << x1 << " , x2=" << x2;
    if(x1 == -2147483648)
        cout << "No real root";
    if(x1 == -2147483648)
        return 0;
    if(x1 == x2)
        cout << "Two same roots x=" << x1;
    return 0;
}
