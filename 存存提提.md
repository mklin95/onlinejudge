#include<iostream>
#include<cmath>
using namespace std;

int main()
{
    int nin, tin, i, n, t, m, acc[100];
    
    cin >> nin >> tin;
    
    for(n=0;n<nin;n++)
        acc[n] = 0;
    
    for(i=0;i<tin;i++)
    {
        cin >> n >> t >> m;
        if(t == 1)
            acc[n] = acc[n] + m;
        else if(t == 2)
            acc[n] = acc[n] - m;
    }

    for(i=0;i<nin;i++)
        cout << i << ":" << acc[i] << endl;

    return 0;
}
