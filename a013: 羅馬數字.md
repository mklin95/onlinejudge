#include<bits/stdc++.h>
using namespace std;

int main()
{
    string a, b;

    while(cin >> a >> b)
    {
        int x[8] = {0}, y[8] = {0};
        char s[7] = {'I','V','X','L','C','D','M'};
        int val[7] = {1,5,10,50,100,500,1000};
        int n = a.length(), m = b.length();

        if(a[0] == '#') break;


        //char to int
        for(int i = 0; i < n; i++)
            for(int j = 0; j < 7; j++)
                if(a[i] == s[j])
                {
                    x[i] = val[j];
                    break;
                }
        for(int i = 0; i < m; i++)
            for(int j = 0; j < 7; j++)
                if(b[i] == s[j])
                {
                    y[i] = val[j];
                    break;
                }

        int ans1 = 0, ans2 = 0;

        //count sum
        for(int i = n-1; i >= 0; i--)
        {
            if(x[i] > x[i-1]) //ex:IX=10-1
            {
                ans1 = ans1 + (x[i] - x[i-1]);
                i--;
            }
            else if(x[i] <= x[i-1]) ans1 += x[i]; //ex:II=1+1
        }
        for(int i = m-1; i >= 0; i--)
        {
            if(y[i] > y[i-1])
            {
                ans2 = ans2 + (y[i] - y[i-1]);
                i--; //一次兩格了
            }
            else if(y[i] <= y[i-1]) ans2 += y[i];
        }
        //cout << ans1 << endl << ans2 << endl;
        int ans = 0;
        ans = abs(ans1 - ans2);

        //digit to rome
        if(ans == 0) cout << "ZERO" << endl;

        string ch[] = {"I","IV","V","IX","X","XL","L","XC","C","CD","D","CM","M"};
        int chi[14] = {1,4,5,9,10,40,50,90,100,400,500,900,1000};

        for(int i = 12; i >= 0; i--)
        {
            int n = ans / chi[i];

            for(int j = 0; j < n; j++) cout << ch[i];

            ans = ans % chi[i];
        }

        cout << endl;
    }

    return 0;
}
