#include<bits/stdc++.h>
using namespace std;

int main()
{
    string a, b;
    int x[8], y[8];
    char s[7] = {'I','V','X','L','C','D','M'};
    int val[7] = {1,5,10,50,100,500,1000};

    while(cin >> a >> b)
    {
        if(a[0] == '#') break;

        int n = a.length(), m = b.length();

        for(int i = 0; i < n; i++)
            for(int j = 0; j < 7; j++)
                if(a[i] == s[j])
                {
                    x[i] = val[j];
                    break;
                }
        for(int i = 0; i < m; i++)
            for(int j = 0; j < 7; j++)
                if(b[i] == s[j])
                {
                    y[i] = val[j];
                    break;
                }

        int ans1 = 0, ans2 = 0;

        for(int i = n-1; i >= 0; i--)
        {
            if(x[i] > x[i-1])
            {
                ans1 = ans1 + (x[i] - x[i-1]);
                i--;
            }
            else if(x[i] <= x[i-1]) ans1 += x[i];
        }
        for(int i = m-1; i >= 0; i--)
        {
            if(y[i] > y[i-1])
            {
                ans2 = ans2 + (y[i] - y[i-1]);
                i--;
            }
            else if(y[i] <= y[i-1]) ans2 += y[i];
        }
        int ans = ans1 - ans2;

        cout << ans << endl;


    }

    return 0;
}
